# Configuration SMTP pour l'envoi d'emails
# Copiez ces lignes et exécutez-les dans votre terminal Windows PowerShell

# Configuration Gmail (recommandé pour le développement)
$env:SMTP_HOST = "smtp.gmail.com"
$env:SMTP_PORT = "587"
$env:SMTP_USER = "votre-email@gmail.com"
$env:SMTP_PASSWORD = "votre-mot-de-passe-app"
$env:SMTP_FROM = "votre-email@gmail.com"
